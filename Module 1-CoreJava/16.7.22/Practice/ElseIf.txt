

public class ElseIf {
public static void main(String[] args) {
	int marks=85;
	if(marks>=70)
	{
		System.out.println("A grade");
	}
	else if(marks>=60){
		System.out.println("B grade");
	}
	else if(marks>=50)
	{
		System.out.println("C grade");
	}
	else if(marks>=40)
	{
		System.out.println("D grade");
	}
	else {
		System.out.println("Fail");
	}
}
}
